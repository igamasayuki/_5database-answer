-- 確認用SQL
-- 集約関数
SELECT
 count(*),count(age),sum(age),avg(age),max(age),min(age),count(gender),count(department_name)
FROM
 employees;

-- GROUP BY
SELECT
 count(*),count(age),sum(age),avg(age),max(age),min(age),count(gender),count(department_name),department_name
FROM
 employees
GROUP BY
 department_name;

-- 40歳を超える人を出力
SELECT
 id, name, age , gender, department_name
FROM
 employees
WHERE
 age > 40;

-- 開発部の人を出力
SELECT
 id, name, age , gender, department_name
FROM
 employees
WHERE
 department_name = '開発部';

-- 部署の平均年齢が40を超える部署を出力
SELECT
 count(*),count(age),sum(age),avg(age),max(age),min(age),count(gender),count(department_name),department_name
FROM
 employees
GROUP BY
 department_name
HAVING
 avg(age) > 40;

 -- 開発部の人を出力
 SELECT
 count(*),count(age),sum(age),avg(age),max(age),min(age),count(gender),count(department_name),department_name
FROM
 employees
GROUP BY
 department_name
HAVING
 department_name = '開発部';

 -- 開発部の人を出力(検索スピードは遅い)
SELECT
 count(*),count(age),sum(age),avg(age),max(age),min(age),count(gender),count(department_name),department_name

 department_name = '開発部'

 department_name;
